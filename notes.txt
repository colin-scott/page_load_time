--------------- Workflow ---------------------

./data/fetch_har_files.rb /path/to/httparchive_pages.csv
./analysis/modify_all_hars.sh data/har/*har
./analysis/convert_all_hars.sh data/har/*har
for F in data/wpr/*
  sudo ./analysis/wpr/replay.py --use_server_delays $F
  # Load webpage
done
